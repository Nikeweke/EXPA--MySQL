CREATE TRIGGER update_emp  -- –Ω–∞–∑–≤. —Ç—Ä–∏–≥–µ—Ä–∞
ON Employees -- –Ω–∞ –∫–∞–∫—É—é —Ç–∞–±–ª–∏—Ü—É —Å—Ç–∞–≤–∏—Ç—å—Å—è
FOR UPDATE -- –Ω–∞ –∫–∞–∫–æ–µ –¥–µ–π—Å—Ç–≤–∏–µ
AS
BEGIN

  DECLARE @FIO NVARCHAR(150),
          @DATE_IN DATETIME

  SET @DATE_IN = GETDATE()

  IF UPDATE(NAME)
   BEGIN
    INSERT INTO EMO_LOG (FIO,DATE_IN) VALUES('UPDATED', @DATE_IN)
   END
END

GO
